%-------------------------------------------------------------------------------
% CONFIGURATIONS
%-------------------------------------------------------------------------------

% Document class and package declaration
\documentclass[12pt,busletter,addrfromright,orderfromdateto]{newlfm}
% Uses hyperref for the email link (removing the styling)
\usepackage[hidelinks]{hyperref}
% Uses datetime2 for the formatting of the date field
\usepackage[en-GB]{datetime2}
% Uses lipsum to generate dummy text
\usepackage{lipsum}

% Remove ordinal (i.e. st, nd, rd) from dates
\DTMlangsetup[en-GB]{ord=omit}

% Set font
\renewcommand{\familydefault}{phv}

% Set up spacing and margins for document
\newlfmP{Headlinewd=0pt,Footlinewd=0pt}
\newlfmP{headermarginsize=36pt}
\newlfmP{addrfromskipafter=30pt}
\newlfmP{dateskipafter=30pt}
\newlfmP{addrtoskipafter=30pt}
\newlfmP{sigsize=40pt}
\newlfmP{sigskipafter=50pt}
\newlfmP{postsigskipbefore=0pt}
\newlfmP{postsigskipafter=0pt}

%-------------------------------------------------------------------------------
% LETTER INFORMATION
%-------------------------------------------------------------------------------

% From name and address
\namefrom{John Smith}
\addrfrom{%
    John Smith\\
    123 Smith Street\\
    TOWN STATE POSTCODE
}

% Date to be displaye. Can replace `\today' with text
\dateset{\today}

% To name and address
\addrto{%
    Fred Williams\\
    Acme Ltd Pty\\
    456 Smith Street\\
    TOWN STATE POSTCODE
}

% Greeting and sign off text
\greetto{To Whom It May Concern,}
\closeline{Respectfully yours,}

%-------------------------------------------------------------------------------
% LETTER CONTENT
%-------------------------------------------------------------------------------

\begin{document}
\begin{newlfm}
	
	% Letter content starts here
	
	\lipsum[1]
	
	\lipsum[2]
	
	Here is a list;
	\begin{itemize}
		\item Item 1.
		\item Item 2.
		\item Item 3.
	\end{itemize}
	
	\lipsum[3]
	
	Email me at \href{mailto:123@email.com}{123@email.com}.
	
	% Letter content ends here. Signature block is inserted here.
	
	% Enclosure and CC list. Comment/delete if not required.
	\encllist{(2) Academic Transcript, Resume}
	\cclist{Nathan Apple, Bob Frog}
	
\end{newlfm}
\end{document}

% End document
